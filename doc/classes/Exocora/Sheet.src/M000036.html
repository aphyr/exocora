    def run
      begin
        # Cast parameters
        @params = cast_params @cgi.params

        # Check parameters
        validate

        # Run process
        context = process

        # Render output
        render context
      rescue
        # Handle errors
        output "<html><head><title>Exocora Error</title></head><body>"
        output "<h1>#{$!.class}</h1>"

        output $!.to_html

        output "</body></html>"
      end
    end
